; Macro de Inscription Febblemind by MDio
; ================================================
; - Materiais: Nightshade, Ginseng e Blank Scrolls
; - Medita quando estiver em 15 de mana
; ================================================
; v0.2

SET %manaRate = 15

; MSGS
IF ! %isWelcomeMsgSent 1
{
  EVENT sysmessage Iniciando Macro de Inscription
  EVENT sysmessage # Desenvolvido por MDio - TFG
  WAIT 2s
}

inicio:

SET %isMediting 0
SET %isWelcomeMsgSent 1

IF #Mana < %manaRate
{
  EVENT sysmessage Mana menor que %manaRate
  GOTO meditation
}
ELSE
{
  ; Find Blank Scroll
  FINDITEM MMF
  IF #FINDCNT > 0
  {
    SET #LOBJECTID #FINDID
    EVENT MACRO 17 0
    WAIT 10
    
    ; Click primeiro item do menu
    CLICK 62 72 d
    WAIT 10

    ; Click primeiro item do menu
    CLICK 62 72 d
    WAIT 1s
  }
  ELSE
  {
    EVENT sysmessage Blank Scrolls nao encontrados
    RETURN
  }
}

meditation:
WHILE #Mana < 100
{
  ; Limpa journal
  DELETEJOURNAL
  
  ; Skill meditation
  IF ! %isMediting = 1
  {
    EVENT sysmessage Iniciando Meditation
    EVENT Macro 13 46
  }
  
  ; Set isMediting true
  SET %isMediting = 1
  
  ; Verificando journal
  IF You_are_at_peace. in #Journal
  {
    EVENT Meditação Completa
    SET %isMediting = 0
    GOTO inicio
  }
}
